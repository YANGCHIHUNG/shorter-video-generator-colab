def ensure_directories_exist(*dirs: str) -> None: """ Create directories if they do not exist. :param dirs: one or more directory paths """ for path in dirs: os.makedirs(path, exist_ok=True)

Media Processing

def convert_mp4_to_mp3( video_path: str, bitrate: str = '192k') -> str: """Extract MP3 from MP4 using ffmpeg.""" audio_path = video_path.rsplit('.',1)[0] + '.mp3' subprocess.run(['ffmpeg', '-i', video_path, '-b:a', bitrate, audio_path]) return audio_path def transcribe_audio( audio_path: str, model_size: str = 'small') -> str: """Transcribe audio to text using Whisper.""" model = whisper.load_model(model_size) result = model.transcribe(audio_path) return result['text']
